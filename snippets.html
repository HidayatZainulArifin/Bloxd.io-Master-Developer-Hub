1. Anti-Void Teleport Teleports players back to spawn if they fall below a certain height (Y level). 
tick = (ms) => {
    const ids = api.getPlayerIds();
    for (const id of ids) {
        const pos = api.getPosition(id);
        /* If player falls below Y = -10 */
        if (pos[1] < -10) {
            api.setPosition(id, 0, 5, 0); /* Teleport to 0, 5, 0 */
            api.sendMessage(id, "Don't fall into the void!");
        }
    }
}
2. Auto-Heal Zone Heals players when they stand on a specific coordinate (e.g., your base). 
onBlockStand = (playerId, x, y, z, blockName) => {
    /* Replace with your base coordinates */
    if (x == 10 && z == 10) {
        const currentHealth = api.getHealth(playerId);
        if (currentHealth < 100) {
            api.applyHealthChange(playerId, 5);
            api.sendFlyingMiddleMessage(playerId, ["+5 HP (Healing Zone)"], 100);
        }
    }
}
3. Custom Kill Message Broadcasts a funny or cool message when a player gets a kill. 
onPlayerKilledOtherPlayer = (attacker, victim) => {
    const nameA = api.getEntityName(attacker);
    const nameV = api.getEntityName(victim);
    api.broadcastMessage(nameA + " sent " + nameV + " to the lobby!", { color: "red" });
    api.playSound(attacker, "levelup", 1, 1);
}
